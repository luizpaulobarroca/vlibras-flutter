```dart
abstract class VLibrasPlatform {
  Future<void> initialize();
  Future<void> translate(String text);
  Future<void> pause();
  Future<void> stop();
  Future<void> resume();
  Future<void> repeat();
  Future<void> setSpeed(double speed);
  void dispose();  // synchronous — ChangeNotifier.dispose() is sync
}
```
